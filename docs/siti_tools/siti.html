215        """Calculate SI of a frame
216
217        Args:
218            frame_data (ndarray): 2D array of input frame data
219
220        Returns:
221            float: SI
222        """
223        # TODO: check array dimensions
224
225        # calculate horizontal/vertical operators
226        sob_x = ndimage.sobel(frame_data, axis=0)
227        sob_y = ndimage.sobel(frame_data, axis=1)
228
229        # crop output to valid window, calculate gradient magnitude
230        si = np.hypot(sob_x, sob_y)[1:-1, 1:-1].std()
231        return si

255    @staticmethod
256    def eotf_1886(
257        frame_data: np.ndarray,
258        gamma: float = DEFAULT_GAMMA,
259        l_min: float = 0.0,
260        l_max: float = 1.0,
261    ) -> np.ndarray:
262        """
263        Electro-optical transfer function defined in ITU-R Rec. BT.1886
264
265        Args:
266            frame_data: pixel values expected in the range [0, 1]
267            gamma: exponent
268            l_min: minimum luminance
269            l_max: maximum luminance
270        Note:
271            As the mapping to display luminance happens outside of this function, l_min and l_max are set to 0.0 and 1.0,
272            respectively. If you change these values while using this function in the context of SI/TI calculation, your
273            output will be wrong. You can however change the values when using this function in a standalone fashion.
274
275        Returns:
276            frame_data: pixel values in the range [0, 1]
277
278        """
279        if np.min(frame_data) < 0:
280            logger.warn("Input for eotf_1886() was < 0, truncating")
281        if np.max(frame_data) > 1:
282            logger.warn("Input for eotf_1886() was > 1, truncating")
283
284        frame_data = np.maximum(frame_data, 0.0)
285        frame_data = np.minimum(frame_data, 1.0)
286
287        a = np.power(np.power(l_max, 1.0 / gamma) - np.power(l_min, 1.0 / gamma), gamma)
288        b = np.power(l_min, 1.0 / gamma) / (
289            np.power(l_max, 1.0 / gamma) - np.power(l_min, 1.0 / gamma)
290        )
291
292        frame_data = a * np.power(np.maximum(frame_data + b, 0), gamma)
293
294        return frame_data

296    @staticmethod
297    def eotf_inv_srgb(frame_data: np.ndarray) -> np.ndarray:
298        """
299        Inverse sRGB electro-optical transfer function
300
301        Args:
302            frame_data: pixel values expected in the range [0, 1]
303
304        Returns:
305            frame_data: pixel values in the range [0, 1]
306
307        """
308        if np.min(frame_data) < 0:
309            logger.warn("Input for eotf_inv_srgb() was < 0, truncating")
310        if np.max(frame_data) > 1:
311            logger.warn("Input for eotf_inv_srgb() was > 1, truncating")
312
313        frame_data = np.maximum(frame_data, 0.0)
314        frame_data = np.minimum(frame_data, 1.0)
315
316        frame_data = (frame_data <= 0.04045) * frame_data / 12.92 + (
317            frame_data > 0.04045
318        ) * np.power((frame_data + 0.055) / 1.055, 2.4)
319
320        return frame_data

322    @staticmethod
323    def apply_display_model(
324        frame_data: np.ndarray,
325        eotf_function: EotfFunction = EotfFunction.BT1886,
326        l_max: float = DEFAULT_L_MAX,
327        l_min: float = DEFAULT_L_MIN,
328        gamma: float = DEFAULT_GAMMA,
329    ) -> np.ndarray:
330        """
331        Apply the SDR EOTF
332
333        Args:
334            frame_data (np.ndarray): Raw frame data in full range, values in the range [0, 1]
335            Other arguments: see calculate()
336
337        Returns:
338            frame_data: pixel values in the physical domain.
339        """
340        if np.min(frame_data) < 0:
341            raise RuntimeError("Input for apply_display_model() was < 0")
342        if np.max(frame_data) > 1:
343            raise RuntimeError("Input for apply_display_model() was > 1")
344
345        if eotf_function == EotfFunction.BT1886:
346            fn = SiTiCalculator.eotf_1886
347            # note that l_min/l_max are not passed here on purpose, as they are set to 0.0 and 1.0,
348            # and mapped outside of the eotf_1886 function
349            kwargs = {"gamma": gamma}
350        elif eotf_function == EotfFunction.INV_SRGB:
351            fn = SiTiCalculator.eotf_inv_srgb
352            kwargs = {}
353        else:
354            raise RuntimeError("Unknown EOTF function!")
355
356        return (l_max - l_min) * fn(frame_data, **kwargs) + l_min

358    @staticmethod
359    def oetf_pq(frame_data: np.ndarray) -> np.ndarray:
360        """
361        Perceptual quantizer opto-electrical transfer function defined in ITU-R Rec. BT.2100
362
363        Args:
364            frame_data: pixel values expected to be in the physical luminance between 0 and 10,000 cd/m^2
365
366        Returns:
367            frame_data: pixel values in the range [0, 1]
368        """
369        m = 78.84375
370        n = 0.1593017578125
371        c1 = 0.8359375
372        c2 = 18.8515625
373        c3 = 18.6875
374        lm1 = np.power(10000.0, n)
375        # FIXME: this might return an error if input is negative, see https://stackoverflow.com/q/45384602/
376        lm2 = np.power(frame_data, n)
377
378        frame_data = np.power((c1 * lm1 + c2 * lm2) / (lm1 + c3 * lm2), m)
379
380        return frame_data

382    @staticmethod
383    def oetf_pu21(
384        frame_data: np.ndarray,
385        mode: Pu21Mode = Pu21Mode.BANDING,
386    ) -> np.ndarray:
387        """
388        PU21 OETF, see https://github.com/gfxdisp/pu21
389
390        Args:
391            frame_data (np.ndarray): Raw frame data in full range, values in the range [0, 1]
392            mode (Pu21Mode), defaults to Pu21Mode.BANDING
393
394        Returns:
395            frame_data: pixel values in the range [0, 1]
396        """
397
398        if mode == "banding":
399            p = [
400                1.070275272,
401                0.4088273932,
402                0.153224308,
403                0.2520326168,
404                1.063512885,
405                1.14115047,
406                521.4527484,
407            ]
408            p_min = -1.5580e-07
409            p_max = 520.4673
410        elif mode == "banding_glare":
411            p = [
412                0.353487901,
413                0.3734658629,
414                8.277049286e-05,
415                0.9062562627,
416                0.09150303166,
417                0.9099517204,
418                596.3148142,
419            ]
420            p_min = 5.4705e-10
421            p_max = 595.3939
422        elif mode == "peaks":
423            p = [
424                1.043882782,
425                0.6459495343,
426                0.3194584211,
427                0.374025247,
428                1.114783422,
429                1.095360363,
430                384.9217577,
431            ]
432            p_min = 1.3674e-07
433            p_max = 380.9853
434        elif mode == "peaks_glare":
435            p = [
436                816.885024,
437                1479.463946,
438                0.001253215609,
439                0.9329636822,
440                0.06746643971,
441                1.573435413,
442                419.6006374,
443            ]
444            p_min = -9.7360e-08
445            p_max = 407.5066
446        else:
447            raise RuntimeError("Invalid mode specified for PU21")
448
449        # scale frame_data to the range 0.005 - 10,000
450        # disabled since the input is already in physical luminance
451        # frame_data = frame_data * 10000.0
452        # frame_data = np.maximum(frame_data, 0.05)
453
454        frame_data = p[6] * (
455            np.power(
456                (
457                    (p[0] + p[1] * np.power(frame_data, p[3]))
458                    / (1.0 + p[2] * np.power(frame_data, p[3]))
459                ),
460                p[4],
461            )
462            - p[5]
463        )
464
465        # get the output into the range [0, 1]
466        frame_data = (frame_data - p_min) / (p_max - p_min)
467
468        return frame_data

470    @staticmethod
471    def eotf_hlg(
472        frame_data: np.ndarray,
473        l_min: float = DEFAULT_L_MIN_HDR,
474        l_max: float = DEFAULT_L_MAX_HDR,
475    ) -> np.ndarray:
476        """
477        Hybrid log-gamma opto-electrical transfer function defined in ITU-T Rec. BT.2100
478
479        Args:
480            frame_data: pixel values expected to be in the range [0, 1]
481
482        Returns:
483            frame_data: pixel values in the physical luminance up to 10,000 cd/m^2
484        """
485
486        a = 0.17883277
487        b = 0.02372241
488        c = 1.00429347
489        if l_max <= 1000.0:
490            gamma = 1.2
491        else:
492            gamma = 1.2 + 0.42 * np.log10(l_max/1000.0)
493
494        frame_data = (frame_data <= 0.5) * (np.power(frame_data, 2.0) / 3.0) + (
495            frame_data > 0.5
496        ) * (np.exp((frame_data - c) / a) - b)
497
498        # NOTE: The following is only valid when the function is applied to the Y channel.
499        # If frame_data was a color channel (R, G, or B), the correct implementation would be:
500        # (l_max - l_min) * np.power(Y, gamma - 1.0) * frame_data + l_min,
501        # where Y would be obtained from linearized R, G, B channels.
502        frame_data = (l_max - l_min) * np.power(frame_data, gamma) + l_min
503
504        return frame_data

536    def handle_limited_range(self, frame_data) -> np.ndarray:
537        """
538        Transforms data from limited range to full range. We expect the input to be in [16/255, 235/255] already.
539        Apply this to limited range content only! If we exceed the limited range, an error will be raised.
540
541        Args:
542            frame_data (np.ndarray): The frame data in limited range.
543
544        Returns:
545            frame_data: The frame data in full range between [0, 1].
546        """
547        input_min = np.min(frame_data)
548        input_max = np.max(frame_data)
549        if (input_min + 0.001 < self.LIMITED_RANGE_MIN) or (
550            input_max - 0.001 > self.LIMITED_RANGE_MAX
551        ):
552            # inform the user about the original range
553            raise RuntimeError(
554                "Input appears to be full range, but it is treated as limited range SDR! "
555                f"Input range is [{int(self.normalize_to_original_range(input_min))}-{int(self.normalize_to_original_range(input_max))}]. "
556                f"Expected range for limited content [{int(self.normalize_to_original_range(self.LIMITED_RANGE_MIN))}-{int(self.normalize_to_original_range(self.LIMITED_RANGE_MAX))}]. "
557                "Specify the range as full instead."
558            )
559        # scale up to full range
560        frame_data = np.clip(
561            (frame_data - self.LIMITED_RANGE_MIN)
562            / (self.LIMITED_RANGE_MAX - self.LIMITED_RANGE_MIN),
563            0,
564            1,
565        )
566
567        return frame_data

575    def calculate(
576        self,
577        input_file: str,
578        num_frames=0,
579    ) -> Tuple[List[float], List[Union[float, None]], int]:
580        """Calculate SI and TI from an input file.
581
582        Returns two arrays and one integer, the first array being the SI values, the
583        second array being the TI values. The first element of the TI value array will
584        always be 0. The integer will be equal to the number of frames read.
585
586        You can also get the results later via .get_results()
587
588        Args:
589            input_file (str): input file path
590            num_frames (int): number of frames to calculate. Defaults to 0 (unlimited).
591
592        Returns:
593            [[float], [float|None], int]: [si_values], [ti_values], frame count
594        """
595        self.si_values = []
596        self.ti_values = []
597        self.last_input_file = input_file
598        previous_frame_data = None
599
600        current_frame = 0
601        for frame_data in read_container(input_file):
602            self.verbose and logger.debug(f"Frame {current_frame+1}")
603
604            if current_frame == 0:
605                logger.debug("Original frame data")
606                self._log_frame_data(frame_data)
607
608            # Normalize frame data according to bit depth between 0 and 1.
609            # This will transform [0, 255] to [0, 1], and [0, 1023] to [0, 1] etc.
610            if not self.legacy:

451        # disabled since the input is already in physical luminance
452        # frame_data = frame_data * 10000.0
453        # frame_data = np.maximum(frame_data, 0.05)

455        frame_data = p[6] * (
456            np.power(

458                    (p[0] + p[1] * np.power(frame_data, p[3]))
459                    / (1.0 + p[2] * np.power(frame_data, p[3]))
460                ),
461                p[4],
462            )
463            - p[5]

472    def eotf_hlg(
473        frame_data: np.ndarray,
474        l_min: float = DEFAULT_L_MIN_HDR,
475        l_max: float = DEFAULT_L_MAX_HDR,
476    ) -> np.ndarray:
477        """
478        Hybrid log-gamma opto-electrical transfer function defined in ITU-T Rec. BT.2100
479
480        Args:
481            frame_data: pixel values expected to be in the range [0, 1]
482
483        Returns:
484            frame_data: pixel values in the physical luminance up to 10,000 cd/m^2
485        """
486
487        a = 0.17883277
488        b = 0.02372241
489        c = 1.00429347
490        if l_max <= 1000.0:
491            gamma = 1.2
492        else:
493            gamma = 1.2 + 0.42 * np.log10(l_max/1000.0)
494
495        frame_data = (frame_data <= 0.5) * (np.power(frame_data, 2.0) / 3.0) + (
496            frame_data > 0.5
497        ) * (np.exp((frame_data - c) / a) - b)
498
499        # NOTE: The following is only valid when the function is applied to the Y channel.
500        # If frame_data was a color channel (R, G, or B), the correct implementation would be:
501        # (l_max - l_min) * np.power(Y, gamma - 1.0) * frame_data + l_min,
502        # where Y would be obtained from linearized R, G, B channels.
503        frame_data = (l_max - l_min) * np.power(frame_data, gamma) + l_min
504
505        return frame_data
